#include<iostream>
using namespace std;
class String
{  public:
	String(const char* str = "")
	{
		size_t len = strlen(str);
		_str = new char[len + 1];
		strcpy(_str, str);
		_size = len;
		_capacity = len;
	}
	String(const String& s)
		:_str(NULL)
	{
		String tmp(s._str);
		tmp.Swap(*this);
	}

	String& operator=(String s)
	{
		this->Swap(s);
		return *this;
     }
	~String()
	{
		if (_str)
		{  
			delete[] _str;
			_str = NULL;
			_size = _capacity = 0;
		}
	}
	char* GetStr()
	{ 
		return _str;
	}
	const char*GetStr() const
	{
		return _str;
	}

	size_t Size()
	{
		return _size;
	}
	size_t Size() const
	 {
		return _size;
	}
	size_t Capacity()
	{
		return _capacity;
	}

	void Expend(size_t  n)
	{
		assert(n > _capacity);
		_str = (char*)realloc(_str, n);
